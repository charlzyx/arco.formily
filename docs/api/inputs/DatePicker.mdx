# DatePicker

> 日期选择器

## Markup Schema 案例

```tsx
import React from "react";
import {
  DatePicker,
  FormItem,
  FormButtonGroup,
  Submit,
} from "arco.formily";
import { createForm } from "@formily/core";
import { FormProvider, createSchemaField } from "@formily/react";

const SchemaField = createSchemaField({
  components: {
    DatePicker,
    FormItem,
  },
});

const form = createForm();

export default () => (
  <React.Fragment>
  <FormProvider form={form}>
    <SchemaField>
      <SchemaField.String
        name="date"
        required
        title="普通日期"
        x-decorator="FormItem"
        x-component="DatePicker"
      />
      <SchemaField.String
        name="week"
        title="周选择"
        x-decorator="FormItem"
        x-component="DatePicker.WeekPicker"
        x-component-props={{
          // picker: "week",
        }}
      />
      <SchemaField.String
        name="month"
        title="月选择"
        x-decorator="FormItem"
        x-component="DatePicker.MonthPicker"
        x-component-props={{
          // picker: "month",
        }}
      />
      <SchemaField.String
        name="quarter"
        title="财年选择"
        x-decorator="FormItem"
        x-component="DatePicker.QuarterPicker"
        x-component-props={{
          // picker: "quarter",
        }}
      />
      <SchemaField.String
        name="year"
        title="年选择"
        x-decorator="FormItem"
        x-component="DatePicker.YearPicker"
        x-component-props={{
          // picker: "year",
        }}
      />
      <SchemaField.String
        name="[startDate,endDate]"
        title="日期范围"
        x-decorator="FormItem"
        x-component="DatePicker.RangePicker"
        x-component-props={{
          showTime: true,
        }}
      />
      <SchemaField.String
        name="range_week"
        title="周范围选择"
        x-decorator="FormItem"
        x-component="DatePicker.RangePicker"
        x-component-props={{
          mode: "week",
        }}
      />
      <SchemaField.String
        name="range_month"
        title="月范围选择"
        x-decorator="FormItem"
        x-component="DatePicker.RangePicker"
        x-component-props={{
          mode: "month",
        }}
      />
      <SchemaField.String
        name="range_quarter"
        title="财年范围选择"
        x-decorator="FormItem"
        x-component="DatePicker.RangePicker"
        x-component-props={{
          mode: "quarter",
        }}
      />
      <SchemaField.String
        name="range_year"
        title="年范围选择"
        x-decorator="FormItem"
        x-component="DatePicker.RangePicker"
        x-component-props={{
          mode: "year",
        }}
      />
    </SchemaField>
    <FormButtonGroup>
      <Submit onSubmit={console.log}>提交</Submit>
    </FormButtonGroup>
  </FormProvider>
  </React.Fragment>

);
```

## JSON Schema 案例

```tsx
import React from "react";
import {
  DatePicker,
  FormItem,
  FormButtonGroup,
  Submit,
  Switch,
} from "arco.formily";
import { FormProvider, createSchemaField } from "@formily/react";
import { ArrayField, createForm, onFieldValueChange } from "@formily/core";
import { ConfigProvider } from '@arco-design/web-react';
const SchemaField = createSchemaField({
  components: {
    DatePicker,
    FormItem,
    Switch,
  },
});

const form = createForm({
  effects: () => {
    onFieldValueChange(".readPretty", (field, form) => {
      form.readPretty = field.value;
      field.readPretty = false;
    });
  },
});

const schema = {
  type: "object",
  properties: {
    readPretty: {
      title: "切换阅读态",
      "x-decorator": "FormItem",
      "x-component": "Switch",
```
